import numpy as np
from PIL import Image

def Encode(src, message, des):
    img = Image.open(src)
    
    if img.mode != 'RGB':
        img = img.convert('RGB')
    width, height = img.size
    array = np.array(img)
    total_pixels = array.size // 3
    # Add delimiter
    message += "$t3g0"
    b_message = ''.join([format(ord(i), "08b") for i in message])
    required_bits = len(b_message)
    if required_bits > total_pixels * 3:
        print("ERROR: Image too small for message")
        return
    index = 0
    # Encode
    for i in range(height):
        for j in range(width):
            for k in range(3):
                if index < required_bits:
                    array[i][j][k] = (array[i][j][k] & 254) | int(b_message[index])
                    index += 1
    # Save as PNG (lossless)
    enc_img = Image.fromarray(array.astype('uint8'))
    enc_img.save(des)
    print("Image Encoded Successfully")

def Decode(src):
    img = Image.open(src)
    
    if img.mode != 'RGB':
        img = img.convert('RGB')
    array = np.array(img)
    height, width, _ = array.shape
    hidden_bits = ""
    # Extract bits
    for i in range(height):
        for j in range(width):
            for k in range(3):
                hidden_bits += str(array[i][j][k] & 1)
    # Convert to characters
    chars = [hidden_bits[i:i+8] for i in range(0, len(hidden_bits), 8)]
    message = ""
    for c in chars:
        message += chr(int(c, 2))
        if message.endswith("$t3g0"):
            print("Hidden Message:", message[:-5])
            return
    print("NO hidden message")

def Stego():
    print("-----Welcome to $t3g0------")
    print("1. Encode")
    print("2. Decode")
    choice = input("Enter your choice: ")
    if choice == '1':
        src = input("Enter source image path: ")
        message = input("Enter message to hide: ")
        des = input("Enter destination image path: ")
        print("Encoding...")
        Encode(src, message, des)
    elif choice == '2':
        src = input("Enter source image path: ")
        print("Decoding...")
        Decode(src)
    else:
        print("Invalid choice")

if __name__ == "__main__":
    Stego()
